## EmailPasswordForm

An email password form (two TextFieldWithHint and a PressableButton).

```kotlin
fun EmailPasswordForm(
    modifier: Modifier = Modifier,
    emailLabel: String = "Email",
    passwordLabel: String = "Password",
    buttonLabel: String = "Login",
    onDone: (String, String) -> Unit
)
```

Example : 
```kotlin
var e by remember { mutableStateOf("") }
var p by remember { mutableStateOf("") }
EmailPasswordForm(
    onDone = { email, password ->
        e = email
        p = password
    },
    modifier = centerHorizontallyModifier,
)
```

## KickstarterTopAppBar

A simple top app bar .

```kotlin
fun KickstarterTopAppBar(
    title: String,
    navigateBack: () -> Unit,
    actions: (@Composable RowScope.() -> Unit)? = null
)
```

Example : 
 
```kotlin 
KickstarterTopAppBar(
    "Detail post", 
    navigateBack = { navController.navigateUp() }, 
    actions = { 
        SimpleIconButton(Icons.Rounded.MoreVert) {}
    }
)
```

## ToggleSetting

A simple toggleable item with title, icon and cta (toggle).

```kotlin 
fun ToggleSetting(
    imageVector: ImageVector,
    title: String,
    subtitle: String,
    modifier: Modifier = Modifier,
    checked: Boolean,
    onCheckedChange: (Boolean) -> Unit
)

```

Example : 
 
```kotlin 
var remindersChecked by remember { mutableStateOf(false) }                 
ToggleSetting(
    title = "Daily reminder notifications",
    subtitle = "Reminders are ${if (remindersChecked) "enabled" else "disabled"}",
    checked = remindersChecked,
    onCheckedChange = { remindersChecked = !remindersChecked },
    modifier = Modifier.padding(horizontal = 16.dp),
    imageVector = Icons.Rounded.Alarm
)
```
